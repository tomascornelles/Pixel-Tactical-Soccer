function obtenerEstiloCamiseta(c1, c2, patron) { if (patron === 'rayas') { return `repeating-linear-gradient(90deg, ${c1}, ${c1} 6px, ${c2} 6px, ${c2} 12px)`; } else if (patron === 'mitad') { return `linear-gradient(90deg, ${c1} 50%, ${c2} 50%)`; } else if (patron === 'diagonal') { return `linear-gradient(135deg, ${c1} 50%, ${c2} 50%)`; } else if (patron === 'h-lineas') { return `repeating-linear-gradient(0deg, ${c1}, ${c1} 6px, ${c2} 6px, ${c2} 12px)`; } else if (patron === 'linea-diagonal') { return `linear-gradient(135deg, transparent 47%, ${c1} 47%, ${c1} 53%, transparent 53%), linear-gradient(to right, ${c2}, ${c2})`; } else if (patron === 'v-linea') { return `linear-gradient(90deg, transparent 45%, ${c1} 45%, ${c1} 55%, transparent 55%), linear-gradient(to bottom, ${c2}, ${c2})`; } else if (patron === 'damero') { const size = '10px'; return ` linear-gradient(45deg, ${c1} 25%, transparent 25%), linear-gradient(-45deg, ${c1} 25%, transparent 25%), linear-gradient(45deg, transparent 75%, ${c1} 75%), linear-gradient(-45deg, transparent 75%, ${c1} 75%); background-size: ${size} ${size}; background-position: 0 0, 0 ${size}, ${size} -${size}, -${size} 0; background-color: ${c2}; `.replace(/\n\s+/g, ''); } else if (patron === 'solido') { // Patrón sólido con línea horizontal en la parte superior return `linear-gradient(180deg, ${c2} 0%, ${c2} 8%, ${c1} 8%, ${c1} 100%)`; } return c1; }
